Named Tensors: CNN / LeNet

Conv 2D

\[\begin{aligned} W &\in \mathbb{R}^{\ensuremath{\ensuremath{\mathsf{channel}}[c]} \times\ensuremath{\ensuremath{\mathsf{kh}}[kh]} \times \ensuremath{\ensuremath{\mathsf{kw}}[kw]}} \\ h' &= h-kh +1, w' = w-kw +1 \\ b &\in \mathbb{R}^{\ensuremath{\ensuremath{\mathsf{height}}[h']} \times \ensuremath{\ensuremath{\mathsf{width}}[w']}} \\ \text{conv2d}(X; W, b) &= W \mathbin{\mathop{\boldsymbol\cdot}\limits_{\ensuremath{\mathsf{channel, kh, kw}}}} U + b \\ U &\in \mathbb{R}^{{\ensuremath{\ensuremath{\mathsf{channel}}[c]} \times \ensuremath{\ensuremath{\mathsf{height}}[h']} \times \ensuremath{\ensuremath{\mathsf{width}}[w']} \times \ensuremath{\ensuremath{\mathsf{kh}}[kh]} \times \ensuremath{\ensuremath{\mathsf{kw}}[kw]}}} \\ U_{\ensuremath{\ensuremath{\mathsf{height}}(i)}, \ensuremath{\ensuremath{\mathsf{width}}(j)}, \ensuremath{\ensuremath{\mathsf{kh}}(i')}, \ensuremath{\ensuremath{\mathsf{kw}}(j')}} &= X_{\ensuremath{\ensuremath{\mathsf{height}}(i+i' - 1)}, \ensuremath{\ensuremath{\mathsf{width}}(j+j' - 1)}} \end{aligned}\]

Max Pooling

\[\begin{aligned} X &\in \mathbb{R}^{\ensuremath{\ensuremath{\mathsf{height}}[h]} \times \ensuremath{\ensuremath{\mathsf{width}}[w]}} \\ \text{maxpool2d}(X, kh, kw) &= \mathop{\mathrm{max}}\limits_{\ensuremath{\mathsf{kh, kw}}} U \\ U &\in \mathbb{R}^{{\ensuremath{\ensuremath{\mathsf{height}}[h / kh]} \times \ensuremath{\ensuremath{\mathsf{width}}[w / kw]} \times \ensuremath{\ensuremath{\mathsf{kh}}[kh]} \times \ensuremath{\ensuremath{\mathsf{kw}}[kw]}}} \\ U_{\ensuremath{\ensuremath{\mathsf{height}}(i)}, \ensuremath{\ensuremath{\mathsf{width}}(j)}, \ensuremath{\ensuremath{\mathsf{kh}}(di)}, \ensuremath{\ensuremath{\mathsf{kw}}(dj)}} & = X_{\ensuremath{\ensuremath{\mathsf{height}}(i \times kh + di -1)}, \ensuremath{\ensuremath{\mathsf{width}}(j \times kw + dj - 1)}} \end{aligned}\]

MLP

\[\begin{aligned} W^1 &\in \mathbb{R}^{\ensuremath{\ensuremath{\mathsf{inp}}[i]} \times \ensuremath{\ensuremath{\mathsf{hidden}}[h]}} \times W^2 \in \mathbb{R}^{ \ensuremath{\ensuremath{\mathsf{hidden}}[h]}} \\ \text{MLP}(X; W, b) &= W^1 \mathbin{\mathop{\boldsymbol\cdot}\limits_{\ensuremath{\mathsf{hidden}}}} \text{ReLU}(W^1 \mathbin{\mathop{\boldsymbol\cdot}\limits_{\ensuremath{\mathsf{inp}}}} X + b^1) + b^2 \end{aligned}\]

Model

\[\begin{aligned} \textit{Params}: \\ W^1 & \in \mathbb{R}^{\ensuremath{\ensuremath{\mathsf{channels'}}[c_2]} \times \ensuremath{\ensuremath{\mathsf{channels}}[c_1]} \times \ensuremath{\ensuremath{\mathsf{kh}}[kh_1]} \times \ensuremath{\ensuremath{\mathsf{kw}}[kw_1]}} \\ W^2 & \in \mathbb{R}^{\ensuremath{\ensuremath{\mathsf{channels'}}[c_3]} \times \ensuremath{\ensuremath{\mathsf{channels}}[c_2]} \times \ensuremath{\ensuremath{\mathsf{kh}}[kh_2]} \times \ensuremath{\ensuremath{\mathsf{kw}}[kw_2]}} \\ U & \in \mathbb{R}^{\ensuremath{\ensuremath{\mathsf{in}}[h_1]} \times \ensuremath{\ensuremath{\mathsf{hidden}}[h_2]}} \times V \in \mathbb{R}^{\ensuremath{\ensuremath{\mathsf{hidden}}[h_2]} \times \ensuremath{\ensuremath{\mathsf{out}}[h_3]}} \\ \\ \textit{Model}: \\ X^0 &\in \mathbb{R}^{\ensuremath{\ensuremath{\mathsf{batch}}[b]} \times \ensuremath{\ensuremath{\mathsf{channels}}[1]} \times \ensuremath{\ensuremath{\mathsf{height}}[h]} \times \ensuremath{\ensuremath{\mathsf{width}}[w]}}\\ X^1 &= \left[\text{ReLU}(\text{conv2d}(X^0; W^1))\right]_{\ensuremath{\mathsf{channels'}}\rightarrow\ensuremath{\mathsf{channels}}} \\ X^2 &= \text{maxpool2d}(X^1, ph_1, ph_2) \\ X^3 &= \left[\text{ReLU}(\text{conv2d}(X^2; W^2))\right]_{\ensuremath{\mathsf{channels'}}\rightarrow\ensuremath{\mathsf{channels}}} \\ X^4 &= \text{maxpool2d}(X^3, ph_2, ph_2) \\ X^5 &\in \mathbb{R}^{\ensuremath{\ensuremath{\mathsf{batch}}[b]} \times \ensuremath{\ensuremath{\mathsf{inp}}[h_1]}}\\ X^5_{\ensuremath{\mathsf{inp}}} &= X^4_{\ensuremath{\mathsf{height}}, \ensuremath{\mathsf{width}}, \ensuremath{\mathsf{channels}}}\\ O &= \mathop{\mathrm{softmax}}\limits_{\ensuremath{\mathsf{class}}} \left[\text{MLP}(X^5; U, V)\right]_{\ensuremath{\mathsf{out}}\rightarrow\ensuremath{\mathsf{class}}}\end{aligned}\]
